USE HOSPITAL;


CREATE TABLE USERID(
STAFF_ID VARCHAR(7), 
STAFF_NAME VARCHAR(30), 
PASSWORD VARCHAR(30), 
HOSPITAL_ID VARCHAR(7), 
AADHAR_NUMBER VARCHAR(13), 
PRIMARY KEY(STAFF_ID, HOSPITAL_ID, AADHAR_NUMBER), 
FOREIGN KEY(HOSPITAL_ID) REFERENCES HOSPITALL(HOSPITAL_ID) ON DELETE CASCADE
);


DESC USERID;

SELECT * FROM USERID;

TRUNCATE TABLE USERID;




CREATE TABLE HOSPITALL(
HOSPITAL_ID VARCHAR(7) PRIMARY KEY
);

DESC HOSPITALL;

SELECT * FROM HOSPITALL;



CREATE TABLE PATIENT(
PATIENT_ID INTEGER PRIMARY KEY,
PATIENT_NAME VARCHAR(50), 
MOBILE_NO LONG, 
AGE VARCHAR(3),
SEX CHAR(1), 
MARITAL_STATUS VARCHAR(15), 
BLOOD_GROUP VARCHAR(5), 
ADDRESS VARCHAR(100) 
);

DESC PATIENT;

SELECT * FROM PATIENT;


CREATE TABLE DONOR(
DONOR_ID INTEGER PRIMARY KEY, 
DONOR_NAME VARCHAR(30),
MOBILE_NO LONG,
AGE INTEGER,
SEX CHAR(1),
MARITAL_STATUS VARCHAR(10),
BLOOD_GROUP VARCHAR(5),
ADDRESS VARCHAR(100)
);

DESC DONOR;

SELECT * FROM DONOR;



CREATE TABLE ORGAN3(
DONOR_ID VARCHAR(10),
ORGAN_ID VARCHAR(5),
ORGAN_NAME VARCHAR(10),
PRIMARY KEY(ORGAN_ID)
);

DESC ORGAN3;

SELECT * FROM ORGAN3;

TRUNCATE TABLE ORGAN3;




CREATE TABLE PREPARE_BILL(
BILL_NO VARCHAR(100),
TREATMENT_CHARGES FLOAT,
BED_CHARGES FLOAT,
OTHER_CHARGES FLOAT,
PATIENT_ID INTEGER,
PRIMARY KEY(BILL_NO, PATIENT_ID),
FOREIGN KEY(PATIENT_ID) REFERENCES PATIENT(PATIENT_ID) ON DELETE CASCADE
);

DESC PREPARE_BILL;

TRUNCATE TABLE PREPARE_BILL;

SELECT * FROM PREPARE_BILL;
